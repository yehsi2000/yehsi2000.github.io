84inline size_t utf8_clamp_to_codepoint(const uint8_t* s, size_t len,
85 size_t maxBytes) {
86 size_t i = 0;
87 while (i < len && i < maxBytes) {
88 uint8_t c = s[i];
89 size_t n = 1;
90 if ((c & 0x80) == 0x00) // 1byte header
91 n = 1;
92 else if ((c & 0xE0) == 0xC0) // 2byte header
93 n = 2;
94 else if ((c & 0xF0) == 0xE0) // 3byte header
95 n = 3;
96 else if ((c & 0xF8) == 0xF0) // 4byte header
97 n = 4;
98 else
99 break; // invalid lead byte
100
101 if (i + n > maxBytes) break; // would split codepoint (by capacity)
102 if (i + n > len) return i; // truncated payload
103 for (size_t k = 1; k < n; ++k) { // validate continuation
104 if ((s[i + k] & 0xC0) != 0x80) return i;
105 }
106 i += n;
107 }
108 return i;
109}
